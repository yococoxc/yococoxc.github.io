RedPacketView.m

#import "RedPacketView.h"
#import "RedPacketViewContent.h"

@interface RedPacketView ()<LQSMenuScrollPageViewSource,LQSMenuScrollPageViewDelegate>

@property(nonatomic,strong) NSArray* array;

@property(nonatomic,strong,readwrite) LQSMenuScrollPageView* menuScrollPageView;

@end

@implementation RedPacketView

- (void) initializeWithView {
    [super initializeWithView];
    
    
    [self lqs__setBackgroundColor:@"#ffffff"];
    
    NSMutableArray* keepArray = [NSMutableArray array];
    [keepArray addObject:@{@"name": @"未使用",@"rp_state": @"unused"}];
    [keepArray addObject:@{@"name": @"已使用",@"rp_state": @"used"}];
    [keepArray addObject:@{@"name": @"已过期",@"rp_state": @"expire"}];
    self.array = keepArray;
    
    
    LQSMenuScrollPageView* menuScrollPageView = [LQSMenuScrollPageView new];
    menuScrollPageView.menu.style = LQSMenuStyleAverageDefault;
    menuScrollPageView.menu.buttonTextSelectedColor = @"#BE00FA";
    menuScrollPageView.menu.scrollLineWidth = 30;
    menuScrollPageView.dataSource = self;
    menuScrollPageView.delegate = self;
    [self.contentView addSubview:menuScrollPageView];
    self.menuScrollPageView = menuScrollPageView;
}


- (void) reloadLayout {
    [super reloadLayout];
    
    self.menuScrollPageView.frame = self.contentView.bounds;
}

#pragma mark - LQSMenuScrollPageViewSource
- (NSUInteger) numberOfViewsInLQSMenuScrollPageView:(LQSMenuScrollPageView *)view {
    if ([self.array isKindOfClass:[NSArray class]]) {
        return [self.array count];
    }
    return 0;
}
- (NSString*) lqsMenuScrollPageView:(LQSMenuScrollPageView *)view textOfMenuAtIndex:(NSUInteger)index {
    NSString* text = nil;
    NSDictionary* dictionary = [self.array objectAtIndex:index];
    if ([dictionary isKindOfClass:[NSDictionary class]]) {
        text = [dictionary objectForKey:@"name"];
    }
    return text;
}

- (UIView*) lqsMenuScrollPageView:(LQSMenuScrollPageView *)view viewOfScrollPageAtIndex:(NSUInteger)index {
    
    RedPacketViewContent* temp = [RedPacketViewContent new];
    return temp;
}

- (void) lqsMenuScrollPageView:(LQSMenuScrollPageView *)view didSelectOrScrollToIndex:(NSInteger)index {
    
    RedPacketViewContent* temp = (RedPacketViewContent*)[view.scrollPageView viewWithIndex:index];
    if ([temp isKindOfClass:[RedPacketViewContent class]] == NO) {
        return;
    }
    
    NSString* rp_state = nil;
    NSDictionary* dictionary = [self.array objectAtIndex:index];
    if ([dictionary isKindOfClass:[NSDictionary class]]) {
        rp_state = [dictionary objectForKey:@"rp_state"];
    }
    
    temp.rp_state = rp_state;
}

@end

#import "LQSView.h"

RedPacketViewContent.m

#import "RedPacketViewContent.h"
#import "MJRefresh.h"
#import "MemberRedpacket__RedpacketListRequest.h"
#import "RedPacketCollectionCell.h"

@interface RedPacketView2Content ()<LQSCollectionViewDataSource,LQSRequestImplementDelegate>
{
    BOOL isRequest;
}

@property(nonatomic,strong) LQSCollectionView* collectionView;

@property(nonatomic,assign) NSUInteger page;

@property(nonatomic,assign) NSInteger page_total;

@property(nonatomic,strong) NSMutableArray* redpacket_list;

@end

@implementation RedPacketView2Content


- (void) initializeWithView {
    [super initializeWithView];
    
    [self lqs__setBackgroundColor:@"#f7f7f7"];
    
    self.redpacket_list = [NSMutableArray array];
    
    RedPacketViewContentEmpty* emptyView = [RedPacketViewContentEmpty new];
    emptyView.hidden = YES;
    self.emptyView = emptyView;
    
    LQSCollectionView* collectionView = [LQSCollectionView new];
    collectionView.collectionView.backgroundView = emptyView;
    collectionView.cellHeight = 92;
    collectionView.dataSource = self;
    collectionView.collectionLayout.minimumLineSpacing = 10;
    collectionView.collectionLayout.sectionInset = UIEdgeInsetsMake(10, 0, 10, 0);
    [collectionView.collectionView registerClass:[RedPacketCollectionCell class] forCellWithReuseIdentifier:@"RedPacketCollectionCell"];
    [self setupBodyView:collectionView];
    self.collectionView = collectionView;
    
    collectionView.collectionView.mj_header = [MJRefreshNormalHeader headerWithRefreshingTarget:self refreshingAction:@selector(reloadRequestAction)];
    
    collectionView.collectionView.mj_footer = [MJRefreshBackNormalFooter footerWithRefreshingTarget:self refreshingAction:@selector(loadRequestAction)];
}


#pragma mark -
- (void) setRp_state:(NSString *)rp_state {
    if (_rp_state != nil) {
        return;
    }
    _rp_state = rp_state;
    
    [self.collectionView.collectionView.mj_header beginRefreshing];
}

#pragma mark - 请求
//重新请求
- (void) reloadRequestAction {
    if (isRequest) {
        return;
    }
    
    self.page = 1;
    self.page_total = -1;
    
    [self actionWithRequest];
}
//请求
- (void) loadRequestAction {
    if (isRequest) {
        return;
    }
    
    [self actionWithRequest];
}


- (void) actionWithRequest {
    if (self.page_total > 0 &&
        self.page >= self.page_total) {
        [self.collectionView.collectionView.mj_header endRefreshing];
        [self.collectionView.collectionView.mj_footer endRefreshing];
        return;
    }
    
    NSString* pageString = [NSString stringWithFormat:@"%d",(int)self.page];
    
    MemberRedpacket__RedpacketListRequest* request = [MemberRedpacket__RedpacketListRequest requestWithDelegate:self];
    [request requestWithRp_state:self.rp_state page:pageString only:&isRequest];
}


#pragma mark - LQSRequestImplementDelegate
- (void) lqsRequestImplement:(LQSRequestImplement *)request error:(NSError *)error {
    NSLog(@"%s  ==  请求失败  ==  %@",__func__, error);
    
    
    [self.collectionView.collectionView.mj_header endRefreshing];
    [self.collectionView.collectionView.mj_footer endRefreshing];
}
- (void) lqsRequestImplement:(LQSRequestImplement *)request success:(id)result {
    NSLog(@"%s  ==  请求成功  ==  %@",__func__, result);
    
    
    [self.collectionView.collectionView.mj_header endRefreshing];
    [self.collectionView.collectionView.mj_footer endRefreshing];
}
- (void) lqsRequestImplement:(LQSRequestImplement *)request successWithDictionary:(NSDictionary *)result code200ForDatasDcitionary:(NSDictionary *)datas {
    if (self.page == 1) {
        self.redpacket_list = [NSMutableArray array];
    }
    
    
    if ([result isKindOfClass:[NSDictionary class]]) {
        NSString* page_total = [result lqs__toStringWithObjectForKey:@"page_total"];
        if (page_total) {
            self.page_total = [page_total integerValue];
        }
    }
    
    if ([datas isKindOfClass:[NSDictionary class]]) {
        NSArray* redpacket_list = [datas objectForKey:@"redpacket_list"];
        if ([redpacket_list isKindOfClass:[NSArray class]]) {
            [self.redpacket_list addObjectsFromArray:redpacket_list];
            if ([redpacket_list count] > 0) {
                self.page++;
            }
        }
    }
    
    if ([self.redpacket_list count] == 0) {
        self.emptyView.hidden = NO;
        self.collectionView.collectionView.mj_footer.hidden = YES;
    } else {
        self.emptyView.hidden = YES;
        self.collectionView.collectionView.mj_footer.hidden = NO;
    }
    
    
    
    [self.collectionView.collectionView reloadData];
}

#pragma mark - LQSCollectionViewDataSource
- (NSInteger) lqsCollectionView:(LQSCollectionView *)view numberOfItemsInSection:(NSInteger)section {
    if ([self.redpacket_list isKindOfClass:[NSArray class]]) {
        return [self.redpacket_list count];
    }
    return 0;
}
- (UICollectionViewCell*) lqsCollectionView:(LQSCollectionView *)view cellForItemAtIndexPath:(NSIndexPath *)indexPath {
    
    RedPacketCollectionCell* cell = [view.collectionView dequeueReusableCellWithReuseIdentifier:@"RedPacketCollectionCell" forIndexPath:indexPath];
    
    if ([self.rp_state lqs__isEqualToString:@"unused"]) {
        cell.state = RedPacketCellStateUnused;
    } else if ([self.rp_state lqs__isEqualToString:@"used"]) {
        cell.state = RedPacketCellStateUsed;
    } else if ([self.rp_state lqs__isEqualToString:@"expire"]) {
        cell.state = RedPacketCellStateExpire;
    }
    
    NSDictionary* dictionary = [self.redpacket_list objectAtIndex:indexPath.row];
    [cell setupWithDictionary:dictionary];
    
    return cell;
}

@end
