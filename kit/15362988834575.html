#import "ZS__HomeCollectionReusableViewCategory.h"
#import "ZS__HomeCollectionReusableViewCategoryContent.h"

@interface ZS__HomeCollectionReusableViewCategory ()<LQSMenuScrollPageViewSource>

@property(nonatomic,strong) NSArray* dataSource;

@end

@implementation ZS__HomeCollectionReusableViewCategory

- (void) initializeWithView {
    [super initializeWithView];
    
    //读取缓存内容
    NSArray* dataSource = [LQSStore getWithString:@"ecapi.category.list"];
    self.dataSource = dataSource;
    
    self.lqs__backgroundColor = @"#fff";
    
    
    LQSMenuScrollPageView* menuScrollPageView = [LQSMenuScrollPageView new];
    menuScrollPageView.menu.scrollLineWidth = 60;
    menuScrollPageView.dataSource = self;
    [self addSubview:menuScrollPageView];
    [menuScrollPageView mas_makeConstraints:^(MASConstraintMaker *make) {
        make.edges.mas_equalTo(0);
    }];
}

#pragma mark - LQSMenuScrollPageViewSource
- (NSUInteger) numberOfViewsInLQSMenuScrollPageView:(LQSMenuScrollPageView *)view {
    NSUInteger count = [self.dataSource lqs__length];
    if (count < 3) {
        return count;
    }
    return 3;
}

- (NSString* ) lqsMenuScrollPageView:(LQSMenuScrollPageView *)view textOfMenuAtIndex:(NSUInteger)index {
    NSDictionary* dictionary = [self.dataSource objectAtIndex:index];
    if ([dictionary lqs__isNSDictionary]) {
        NSString* name = [dictionary lqs__toStringWithObjectForKey:@"name"];
        return name;
    }
    return @"";
}

- (UIView*) lqsMenuScrollPageView:(LQSMenuScrollPageView *)view viewOfScrollPageAtIndex:(NSUInteger)index {
    ZS__HomeCollectionReusableViewCategoryContent* temp = [ZS__HomeCollectionReusableViewCategoryContent new];
    
    NSDictionary* dictionary = [self.dataSource objectAtIndex:index];
    if ([dictionary lqs__isNSDictionary]) {
        NSArray* categories = [dictionary objectForKey:@"categories"];
        if ([categories lqs__isNSArray]) {
            //数据
            temp.categories = categories;
        }
    }
    return temp;
}

@end

ZS__HomeCollectionReusableViewCategoryContent.m

#import "ZS__HomeCollectionReusableViewCategoryContent.h"

#import "ZS__HomeCollectionReusableViewCategoryContentCell.h"

@implementation ZS__HomeCollectionReusableViewCategoryContent

- (void) initializeWithView {
    [super initializeWithView];
    
    self.lqs__backgroundColor = @"#fff";
    
    
    UIView* line = [UIView lqs__backgroundColor:@"#999"];
    [self.contentView addSubview:line];
    [line mas_makeConstraints:^(MASConstraintMaker *make) {
        make.left.right.top.mas_equalTo(0);
        make.height.mas_equalTo(1);
    }];
}

- (void) setCategories:(NSArray *)categories {
    _categories = categories;
    
    
    if ([categories lqs__isNSArray] == NO) {
        return;
    }
    NSMutableArray* tempKeepArray = [NSMutableArray array];
    NSUInteger index = 0;
    for (NSDictionary* dictioanry in categories) {
        if (index < 8) {
            if ([dictioanry lqs__isNSDictionary]) {
                [tempKeepArray addObject:dictioanry];
                index++;
            }
        }
    }
    
    CGFloat width = [[UIScreen mainScreen] bounds].size.width;
    width = width / 4.0;
    CGFloat height = width + 15;
    
    NSMutableArray* keepArray = [NSMutableArray array];
    LQSCollectionSectionItem* item = [LQSCollectionSectionItem new];
    item.cellWidthMultiple = 4;
    [item setupCellWithClass:[ZS__HomeCollectionReusableViewCategoryContentCell class] array:tempKeepArray height:height];
    [keepArray addObject:item];
    self.collectionDataSource = keepArray;
}

@end

#pragma mark - 处理数据
- (void) setupWithDictionary:(NSDictionary *)dictionary {
    
    self.imageView.image = nil;
    self.titleLabel.text = nil;
    self.theID = nil;
    
    if ([dictionary lqs__isNSDictionary] == NO) {
        return;
    }
    
    if (self.tag == 7) {
        self.titleLabel.text = @"更多";
        return;
    }
    
    self.theID = [dictionary lqs__toStringWithObjectForKey:@"id"];
    
    self.titleLabel.text = [dictionary lqs__toStringWithObjectForKey:@"name"];
    
    
    NSString* thumbs = [dictionary lqs__toStringWithObjectForKey:@"thumbs"];
    if ([thumbs lqs__isNSString]) {
        [self.imageView yy_setImageWithURL:[NSURL URLWithString:thumbs] placeholder:nil];
    }
}

#pragma mark - 触发事件

        return;
    }

    controller.category = self.theID;
    [self lqs__currentNavigationControllerPushViewController:controller];
