#import "Article__ListRequest.h"

@implementation Article__ListRequest

- (void) requestWithPage:(NSString*)page only:(BOOL *)only {
    
    NSString* urlString = [NSString stringWithFormat: @"%@/%@/ecapi.article.list",APIDomainName,APIVersion];
    
    self.coreRequest.HTTPMethod = @"POST";
    
    [self.coreRequest setHTTPBody:@"14" forKey:@"id"];
    
    [self.coreRequest setHTTPBody:@"100" forKey:@"per_page"];
    
    [self.coreRequest setHTTPBody:page forKey:@"page"];
    
    [self requestWithURLString:urlString only:only];
}

ZS__ArticleListView.m

#import "ZS__ArticleListView.h"

#import "Article__ListRequest.h"

#import "MJRefresh.h"

#import "ZS__ArticleListCollectionCell.h"

@interface ZS__ArticleListView ()<LQSRequestImplementDelegate>
{
    BOOL isArticle__ListRequest;
}
@property(nonatomic,assign) BOOL isHeaderRefreshing;
@property(nonatomic,assign) NSUInteger page;
@property(nonatomic,strong) NSMutableArray* keepArray;

@end

@implementation ZS__ArticleListView

- (void) initializeWithView {
    [super initializeWithView];
    
    //初始化数据
    self.keepArray = [NSMutableArray array];
    self.page = 1;
    
    //添加上拉下拉刷新功能
    self.collectionManager.collectionView.mj_header = [MJRefreshNormalHeader headerWithRefreshingTarget:self refreshingAction:@selector(headerWithRefreshing)];
    [self.collectionManager.collectionView.mj_header beginRefreshing];
    self.collectionManager.collectionView.mj_footer = [MJRefreshAutoNormalFooter footerWithRefreshingTarget:self refreshingAction:@selector(footerWithRefreshing)];
}


#pragma mark - 请求
- (void) headerWithRefreshing {
    self.isHeaderRefreshing = YES;
    self.page = 1;
    
    
    [self footerWithRefreshing];
}

- (void) footerWithRefreshing {
    NSString* page = [NSString stringWithFormat:@"%d",(int)self.page];
    
    Article__ListRequest* request = [Article__ListRequest requestWithDelegate:self];
    request.tag = @"Article__ListRequest";
    [request requestWithPage:page only:&isArticle__ListRequest];
}


#pragma mark - LQSRequestImplementDelegate
- (void) lqsRequestImplement:(LQSRequestImplement *)request error:(NSError *)error {
    [self.collectionManager.collectionView.mj_footer endRefreshing];
    [self.collectionManager.collectionView.mj_header endRefreshing];
}
- (void) lqsRequestImplement:(LQSRequestImplement *)request success:(id)result {
    [self.collectionManager.collectionView.mj_footer endRefreshing];
    [self.collectionManager.collectionView.mj_header endRefreshing];
    
    if ([request.tag lqs__isEqualToString:@"Article__ListRequest"]) {
        if ([result lqs__isNSDictionary] == NO) {
            return;
        }
        
        NSString* error_code = [result lqs__toStringWithObjectForKey:@"error_code"];
        if ([error_code lqs__isEqualToString:@"0"] == NO) {
            return;
        }
        
        NSArray* articles = [result objectForKey:@"articles"];
        if ([articles lqs__isNSArray] == NO) {
            return;
        }
        if ([articles count] == 0) {
            return;
        }
        
        self.page++;
        
        if (self.isHeaderRefreshing == YES) {
            self.isHeaderRefreshing = NO;
            [self.keepArray removeAllObjects];
        }
        
        [self.keepArray addObjectsFromArray:articles];
        
        [self setupCollectionViewWithDataSource];
        return;
    }
}

#pragma mark - 数据配置内容的设定
- (void) setupCollectionViewWithDataSource {
    NSMutableArray* keepArray = [NSMutableArray array];
    LQSCollectionSectionItem* item = [LQSCollectionSectionItem new];
    item.sectionInsets = UIEdgeInsetsMake(8, 8, 8, 8);
    item.minimumLineSpacing = 8;
    item.minimumInteritemSpacing = 8;
    item.cellWidthOffset = 16;
    [item setupCellWithClass:[ZS__ArticleListCollectionCell class] array:self.keepArray height:44];
    [keepArray addObject:item];
    self.collectionDataSource = keepArray;
}

@end
